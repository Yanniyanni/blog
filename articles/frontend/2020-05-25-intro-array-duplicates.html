                        # 2. Map

                        缺点：对象无法去重

                        function unique(arr) {
  let map = new Map();
  let array = new Array();  // 数组用于返回结果
  for (let i = 0; i < arr.length; i++) {
    if(map.has(arr[i])) {  // 如果有该key值
      map.set(arr[i], true); 
    } else { 
      map.set(arr[i], false);   // 如果没有该key值
      array .push(arr[i]);
    }
  } 
  return array ;
}

  for(var i = 0; i < arr.length; i++){

          if(arr[i] === arr[j]){
              arr.splice(j,1);
              j--;

                        # 11. 对比

                        方法 优点 缺点
                        Set 代码量少 对象无法去重
                        Map 对象无法去重
                        for + splice 对象无法去重，NaN无法去重
                        for + indexOf 对象无法去重，NaN无法去重
                        for + includes 对象无法去重
                        for + sort 对数字/字母类较有效 需传入特定条件进行排序
                        filter + hasOwnProperty 将object类型都去掉了，仅剩一个
                        filter + indexOf 对象无法去重，不能识别 NaN 和 ”NaN“
                        reduce + includes 对象无法去重
                        lodash.js 工具库，使用方便，完美去重

                        总结

                        以上的方法可以自由组合，除了对象的去重需要特殊处理外，其他的基本都适用。如果数组中掺杂了对象，那么我们只需要判断 对象类型，对对象做特殊的去重处理即可。

    return prev
  },[]); // 设置 prev 默认类型为数组，并且初始值为空的数组

                        function unique(arr){
    let map = new Map();
    for (let item of arr) {
        if (!map.has(item.id)) {
            map.set(item.id, item);
        }
    }
    return [...map.values()];
}

                        # 4. 双重循环

                        var temp = []
function unique(arr){
  arr.forEach(function(a) {
      var check = temp.every(function(b) {
          return a.id !== b.id;
      })
      check ? temp.push(a) : ''
  })
  return temp;
}

                        # 5. 冒泡排序法

                        function unique(arr){
  for (var i = 0; i < arr.length - 1; i++) {
      for (var j = i + 1; j < arr.length; j++) {
          if (arr[i].id == arr[j].id) {
              arr.splice(j, 1);
              j--;
          }
      }
  }
  return arr;
}
