# 获取url中对应的参数

gerUrlParam() {
    const url = location.search
    if (url.indexOf('?') !== -1) {
        const str = url.substr(1)
        const strs = str.split('&')
        for (let i = 0; i < strs.length; i++) {
            strs[i].split('=')[0] = unescape(strs[i].split('=')[1])
            if (strs[i].split('=')[0] === 'searchWord') {
                 return decodeURI(strs[i].split('=')[1])
           }
         }
    }
},

# 一个函数处理多个事件

var btn = document.getElementById('myBtn')
var handler = function(event) {
    switch (event.type){
        case 'click':
            alert('clicked');
            break;
        case 'mouseover':
            event.target.style.backgroundColor = 'red';
            break;
        case 'mouseout':
            event.target.style.backgroundColor = 'blue';
            break;
    }
}
btn.onclick = handler;
btn.onmouseover = handler;
btn.onmouseout = handler;

<!-- 60秒倒计时 -->
<div id="time">60</div>
<script>
    var countdown = 60
    var timer = setInterval(function () {
        if (countdown === 1) {
            clearInterval(timer)
        }
        countdown--
        $('#time').html(countdown)
    }, 1000)
</script>

# 判断是否有安装app，有则打开

web页面打开app的方法：

  1. 直接使用schema协议打开，如<a href="test://" class="mask-fixed-btn">
  2. 使用iframe打开，如下wakeUpApp里的打开android
/*
* 带参打开app
*/

function wakeUpApp(type,download_url){

  if(!isAppEnv()){

    var _type = type || 0;
    console.log("wakeUpApp");
    var _url = "";
    if(isWeixin()){
        window.location.href = (/(Android)/i.test(navigator.userAgent))? download_url.android_url:download_url.ios_url;
    }else{
        if(/(Android)/i.test(navigator.userAgent)){
            _url = getAppUrl("android");
            console.log("安卓打开App对应页面》》》》" + _url);
            //此操作会调起app并阻止接下来的js执行
            var iframe = document.createElement('iframe');
            iframe.src =_url ;
            iframe.target = '';
            document.getElementsByTagName('body')[0].appendChild(iframe);

      //没有安装应用会执行下面的语句
      setTimeout(function(){ window.location.href = download_url.android_url;},1200);
    }else{
      console.log("ios打开App对应页面》》》》" + _url);
      //ios
      window.location = getAppUrl("ios");
      setTimeout(function(){window.location.href = download_url.ios_url;},1200);
    }
  }

   function getAppUrl(t){
        var url = "";
        switch(type){
        case 1: //岗位详情页面
          var job_id = document.getElementById('job_id').value;
          url = t == "android"?"bluecollar://bluecollar.uitype/openJobDetail?value="+job_id:"bluecollar://openJobDetail?value="+job_id
          break;
        default:
          url = t == "android"?"bluecollar://bluecollar.uitype":"bluecollar://";
        }
      return url;
    }
  }
}

# 判断是否是微信

function isWeixin(){ //判断是否是微信
    var ua = navigator.userAgent.toLowerCase();
    if (ua.match(/MicroMessenger/i) == "micromessenger") {
        return true;
    } else {
        return false;
    }
}
