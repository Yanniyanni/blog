<!-- 未知宽高——第一种 使用`table-cell` -->
<div class="table">
    <div class="cell">
        <p>11111</p>
    </div>
</div>

<style>
.table{
    display: table;
    width: 300px;
    height: 300px;
}
.cell {
    display: table-cell;
    vertical-align: middle;
    text-align: center;
}
</style>


<!-- 未知宽高—— 第二种 使用transform -->
<div class="wrap">
    <div class="content"></div>
</div>
<style>
.wrap{
    position: relative;
}
.content{
    position:absolute;
    top:50%;
    left: 50%;
    transform: translate(-50%, -50%);
}
</style>


<!-- 未知宽高——第三种 使用flex -->
<style>
.wrap{
    display: flex;
    align-items: center;/* 垂直居中 */
    justify-content: center;/* 水平居中 */
    width:300px;
    height: 300px;
    background: pink;
}
.container{
    background-color: blue;
}  
</style>
